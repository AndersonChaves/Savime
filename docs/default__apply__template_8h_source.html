125  SavimeResult Apply(string op, DatasetPtr operand1, DatasetPtr operand2,
126  DatasetPtr &destinyDataset) {
127  int numCores = _configurationManager->GetIntValue(MAX_THREADS);
128  int32_t minWorkPerThread =
129  _configurationManager->GetIntValue(WORK_PER_THREAD);
130 
131  int64_t entryCount = operand1->GetEntryCount() <= operand2->GetEntryCount()
132  ? operand1->GetEntryCount()
133  : operand2->GetEntryCount();
134 
135  SET_THREADS(entryCount, minWorkPerThread, numCores);
136 
137  DatasetHandlerPtr op1Handler = _storageManager->GetHandler(operand1);
138  DatasetHandlerPtr op2Handler = _storageManager->GetHandler(operand2);
139 
140  destinyDataset = _storageManager->Create(
141  DataType(SelectType(operand1->GetType(), operand2->GetType(), op), 1),
142  operand1->GetEntryCount());
143  if (destinyDataset == nullptr)
144  throw std::runtime_error("Could not create dataset. ");
145 
146  DatasetHandlerPtr destinyHandler =
147  _storageManager->GetHandler(destinyDataset);
148 
149  auto op1Buffer =
150  BUILD_VECTOR<T1>(op1Handler->GetBuffer(), operand1->GetType());
151 
152  auto op2Buffer =
153  BUILD_VECTOR<T2>(op2Handler->GetBuffer(), operand2->GetType());
154 
155  T3 *destinyBuffer = (T3 *)destinyHandler->GetBuffer();
156 
157  if (op.c_str()[0] == _ADDITION[0]) {
158 #pragma omp parallel
159  for (SubTARPosition i = THREAD_FIRST();i < THREAD_LAST(); ++i)
160  destinyBuffer[i] = (*op1Buffer)[i] + (*op2Buffer)[i];
161  } else if (op.c_str()[0] == _SUBTRACTION[0]) {
162 #pragma omp parallel
163  for (SubTARPosition i = THREAD_FIRST();i < THREAD_LAST(); ++i)
164  destinyBuffer[i] = (*op1Buffer)[i] - (*op2Buffer)[i];
165  } else if (op.c_str()[0] == _MULTIPLICATION[0]) {
166 #pragma omp parallel
167  for (SubTARPosition i = THREAD_FIRST();i < THREAD_LAST(); ++i)
168  destinyBuffer[i] = (*op1Buffer)[i] * (*op2Buffer)[i];
169  } else if (op.c_str()[0] == _DIVISION[0]) {
170 #pragma omp parallel
171  for (SubTARPosition i = THREAD_FIRST();i < THREAD_LAST(); ++i)
172  destinyBuffer[i] = (*op1Buffer)[i] / (*op2Buffer)[i];
173  } else if (op.c_str()[0] == _MODULUS[0]) {
174 #pragma omp parallel
175  for (SubTARPosition i = THREAD_FIRST();i < THREAD_LAST(); ++i)
176  destinyBuffer[i] = fmod((*op1Buffer)[i], (*op2Buffer)[i]);
177  } else if (!op.compare(_POWER)) {
178 #pragma omp parallel
179  for (SubTARPosition i = THREAD_FIRST();i < THREAD_LAST(); ++i)
180  destinyBuffer[i] = pow((*op1Buffer)[i], (*op2Buffer)[i]);
181  } else {
182  throw std::runtime_error("Invalid apply operation.");
183  }
184 
185  op1Handler->Close();
186  op2Handler->Close();
187  destinyHandler->Close();
188  //delete op1Buffer;
189  //delete op2Buffer;
190  return SAVIME_SUCCESS;
191  }
