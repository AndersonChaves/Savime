86 class semaphore {
87 private:
88  mutex mtx;
89  mutex semaphore_mutex;
90  condition_variable cv;
91  condition_variable cv_all;
92  volatile int count;
93  volatile int max;
94  volatile int waiting = 0;
95 
96 public:
97  void setCount(int c) { count = max = c; }
98 
99  void notify() {
100  // unique_lock<mutex> lck(mtx);
101 
102  semaphore_mutex.lock();
103  ++count;
104  --waiting;
105  semaphore_mutex.unlock();
106 
107  cv.notify_one();
108  }
109 
110  void wait() {
111  unique_lock<mutex> lck(mtx);
112  while (count == 0)
113  cv.wait(lck);
114 
115  semaphore_mutex.lock();
116  ++waiting;
117  --count;
118  semaphore_mutex.unlock();
119  }
120 };
