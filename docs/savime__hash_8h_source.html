31 class BaseSavimeHash {
32 public:
33  vector<int64_t> multipliers;
34 
35  virtual inline void calculateMultipliers(vector<DimensionPtr> dimensions) {
36  int32_t numDims = dimensions.size();
37  for (int32_t i = 0; i < numDims; i++) {
38  multipliers.push_back(1);
39  for (int32_t j = i + 1; j < numDims; j++) {
40  int64_t preamble = multipliers[i];
41  multipliers[i] = preamble * dimensions[j]->GetCurrentLength();
42  }
43  }
44 
45  if (numDims == 0) {
46  multipliers.push_back(1.0);
47  }
48  }
49 
50  virtual inline uint64_t GetHash(vector<int64_t> indexes) {
51  uint64_t r = 0;
52  for (int32_t i = 0; i < indexes.size(); i++)
53  r += multipliers[i] * indexes[i];
54  return r;
55  }
56 };
