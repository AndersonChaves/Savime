32 class BaseSavimeHash {
33 public:
34  vector<int64_t> multipliers;
35 
36  virtual inline void calculateMultipliers(vector<DimensionPtr> dimensions) {
37  auto numDims = static_cast<uint32_t>(dimensions.size());
38  for (uint32_t i = 0; i < numDims; i++) {
39  multipliers.push_back(1);
40  for (int32_t j = i + 1; j < numDims; j++) {
41  int64_t preamble = multipliers[i];
42  multipliers[i] = preamble * dimensions[j]->GetCurrentLength();
43  }
44  }
45 
46  if (numDims == 0) {
47  multipliers.push_back(1);
48  }
49  }
50 
51  virtual inline uint64_t GetHash(vector<int64_t> indexes) {
52  uint64_t r = 0;
53  for (int32_t i = 0; i < indexes.size(); i++)
54  r += multipliers[i] * indexes[i];
55  return r;
56  }
57 };
