242 inline void fast_division(int64_t max, int64_t div, int64_t &mul,
243  int64_t &shift) {
244  bool found = false;
245  mul = -1;
246  shift = -1;
247 
248  // zero divider error
249  if (div == 0)
250  return;
251 
252  // this division would always return 0 from 0..max
253  if (max < div) {
254  mul = 0;
255  shift = 0;
256  return;
257  }
258 
259  // catch powers of 2
260  for (int s = 0; s <= 63; s++) {
261  if (div == (1L << s)) {
262  mul = 1;
263  shift = s;
264  return;
265  }
266  }
267 
268  // start searching for a valid mul/shift pair
269  for (shift = 1; shift <= 62; shift++) {
270  // shift factor is at least 2log(div), skip others
271  if ((1L << shift) <= div)
272  continue;
273 
274  // we calculate a candidate for mul
275  mul = (1L << shift) / div + 1;
276 
277  // assume it is a good one
278  found = true;
279 
280  // test if it works for the range 0 .. max
281  // Note: takes too much time for large values of max.
282  if (max < 1000000) {
283  for (int64_t i = max; i >= 1;
284  i--) // testing large values first fails faster
285  {
286  if ((i / div) != ((i * mul) >> shift)) {
287  found = false;
288  break;
289  }
290  }
291  } else {
292  // very fast test, no mathematical proof yet but it seems to work well
293  // test highest number-pair for which the division must 'jump' correctly
294  // test max, to be sure;
295  int64_t t = (max / div + 1) * div;
296  if ((((t - 1) / div) != (((t - 1) * mul) >> shift)) ||
297  ((t / div) != ((t * mul) >> shift)) ||
298  ((max / div) != ((max * mul) >> shift))) {
299  found = false;
300  }
301  }
302 
303  // are we ready?
304  if (found) {
305  break;
306  }
307  }
308 }
