193 inline void fast_division(int64_t max, int64_t div, int64_t &mul,
194  int64_t &shift) {
195  bool found = false;
196  mul = -1;
197  shift = -1;
198 
199  // zero divider error
200  if (div == 0)
201  return;
202 
203  // this division would always return 0 from 0..max
204  if (max < div) {
205  mul = 0;
206  shift = 0;
207  return;
208  }
209 
210  // catch powers of 2
211  for (int s = 0; s <= 63; s++) {
212  if (div == (1L << s)) {
213  mul = 1;
214  shift = s;
215  return;
216  }
217  }
218 
219  // start searching for a valid mul/shift pair
220  for (shift = 1; shift <= 62; shift++) {
221  // shift factor is at least 2log(div), skip others
222  if ((1L << shift) <= div)
223  continue;
224 
225  // we calculate a candidate for mul
226  mul = (1L << shift) / div + 1;
227 
228  // assume it is a good one
229  found = true;
230 
231  // test if it works for the range 0 .. max
232  // Note: takes too much time for large values of max.
233  if (max < 1000000) {
234  for (int64_t i = max; i >= 1;
235  i--) // testing large values first fails faster
236  {
237  if ((i / div) != ((i * mul) >> shift)) {
238  found = false;
239  break;
240  }
241  }
242  } else {
243  // very fast test, no mathematical proof yet but it seems to work well
244  // test highest number-pair for which the division must 'jump' correctly
245  // test max, to be sure;
246  int64_t t = (max / div + 1) * div;
247  if ((((t - 1) / div) != (((t - 1) * mul) >> shift)) ||
248  ((t / div) != ((t * mul) >> shift)) ||
249  ((max / div) != ((max * mul) >> shift))) {
250  found = false;
251  }
252  }
253 
254  // are we ready?
255  if (found) {
256  break;
257  }
258  }
259 }
